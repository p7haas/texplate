%% ----------------------------------------------------------------------------
%%
%% Definiert \textuppercase - Text in GROẞBUCHSTABEN mit Tracking
%%
%% ----------------------------------------------------------------------------
\makeatletter
\newcommand{\textuppercase}[1]{%
  {%
    \addfontfeatures{Numbers={Uppercase,Proportional}}%
    \microtypecontext{tracking=allcaps}%
    \lsstyle\MakeUppercase{#1}}}%
\makeatother
